# Простой запуск (с передачей connection string)
docker build -t studiob2b -f deploy/Dockerfile .
docker run -p 80:8080 \
  -e "ConnectionStrings__MasterDb=Server=host;Database=DB;User=root;Password=pass;" \
  -e "ASPNETCORE_ENVIRONMENT=Production" \
  studiob2b

# С docker-compose (MySQL + App)
cd deploy
cp .env.example .env
# Отредактировать .env
docker-compose up -d
